### 📂 프론트 파일 구조

```text
root
    ├── public
    ├── src
    │     ├── apis                    # API 관련
    │     ├── assets                  # 미디어 파일들(폰트, 아이콘, 이미지..)
    │     │      ├── images           # 이미지 파일 관련
    │     │      └── styles           # 폰트, 색상
    │     │             └── fonts
    │     ├── components              # 컴포넌트
    │     │      └── common           # (공통,페이지별)
    │     ├── routers                 # 라우팅
    │     └── store                   # 상태관리
    │     └── views                   # 어플리케이션 경로의 진입점
    └── dist                          # 빌드 파일
          └── ...
```
